412template<typename T>
413inline T get(const properties& props, property::code propid, size_t idx)
414{
415 MQTTProperty* prop = MQTTProperties_getPropertyAt(
416 const_cast<MQTTProperties*>(&props.c_struct()),
417 MQTTPropertyCodes(propid), int(idx));
418 if (!prop)
419 throw bad_cast();
420
421 return get<T>(property(*prop));
422}
